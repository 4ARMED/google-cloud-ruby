class AliasContext
  # The type of an Alias.
  module Kind
    # Do not use.
    ANY = 0

    # Git tag
    FIXED = 1

    # Git branch
    MOVABLE = 2

    # OTHER is used to specify non-standard aliases, those not of the kinds
    # above. For example, if a Git repo has a ref named "refs/foo/bar", it
    # is considered to be of kind OTHER.
    OTHER = 4
  end
end
