def request_response *args
  Google::Cloud::Trace.in_span SPAN_NAME do |span|
    if span && !args.empty?
      grpc_request = args[0]
      label_key = Google::Cloud::Trace::LabelKey::RPC_REQUEST_TYPE
      span.labels[label_key] = grpc_request.class.name.gsub(/^.*::/, "")
    end

    super
  end
end
